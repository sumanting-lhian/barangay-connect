Design Language

The foundational tokens and shared layout components for BarangayConnect — all defined in a single main.css file. This page covers the full color palette and alpha overlays, the two typefaces and their scale, spacing and border-radius tokens, shadow levels, and the role-aware navbar with its three variants (Resident, Officer, Admin) plus the site footer.

Navbar — Transparent (Hero)

.navbar--transparent

Add .navbar--transparent when the navbar sits over a dark photo or hero section. A scroll listener removes it once the user scrolls past 80px, restoring the white bar. Include the provided JS snippet once per page.

Transparent — white text, no background
BarangayConnect
Home Community Services My Barangay
Resident
↑ Navbar floats over hero content here
<!-- Add class to header; JS removes it on scroll -->
<header class="navbar navbar--transparent" id="mainNavbar">
  ...
</header>

/* JS — paste once before </body> */
(function () {
  const nav = document.getElementById('mainNavbar');
  if (!nav) return;
  const THRESHOLD = 80;
  function update() {
    nav.classList.toggle('navbar--transparent', window.scrollY < THRESHOLD);
  }
  window.addEventListener('scroll', update, { passive: true });
  update();
})();

JS Snippets

copy-paste into your page

These snippets are self-contained and can be pasted just before </body> on any page using the navbar. The role-setter is typically called after Firebase Auth resolves the user's Firestore document.

Full navbar JS — transparent scroll + active link + mobile drawer 
/* ── Navbar JS — paste once before </body> on every page ─────────────── */
(function () {
  'use strict';

  const nav       = document.getElementById('mainNavbar');
  const hamburger = document.querySelector('.navbar__hamburger');
  const drawer    = document.querySelector('.navbar__mobile-drawer');
  const links     = document.querySelectorAll('.navbar__link');

  /* 1. Transparent on scroll */
  if (nav && nav.classList.contains('navbar--transparent')) {
    const THRESHOLD = 80;
    function updateTransparency() {
      nav.classList.toggle('navbar--transparent', window.scrollY < THRESHOLD);
    }
    window.addEventListener('scroll', updateTransparency, { passive: true });
    updateTransparency();
  }

  /* 2. Active link — marks current page */
  const currentPath = window.location.pathname.replace(/\/$/, '') || '/';
  links.forEach(link => {
    const href = link.getAttribute('href').replace(/\/$/, '') || '/';
    if (href === currentPath) link.classList.add('is-active');
  });

  /* 3. Mobile drawer toggle */
  if (hamburger && drawer) {
    hamburger.addEventListener('click', () => {
      const open = drawer.classList.toggle('is-open');
      hamburger.setAttribute('aria-expanded', String(open));
    });
    document.addEventListener('click', e => {
      if (!nav.contains(e.target)) drawer.classList.remove('is-open');
    });
  }

  /* 4. Set role class — call after auth resolves */
  window.setNavRole = function (role) {
    document.body.classList.remove('role-resident', 'role-officer', 'role-admin');
    document.body.classList.add('role-' + role);
    const badge = document.querySelector('.navbar__role');
    if (badge) {
      badge.className = 'navbar__role navbar__role--' + role;
      badge.textContent = role.charAt(0).toUpperCase() + role.slice(1);
    }
  };
})();
